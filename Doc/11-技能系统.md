### 5.2 useSkill() 方法（line 414175）

```csharp
public virtual Boolean useSkill(
    Int32 skillIndex, 
    ENUM_DIRECTION_MOBILE forceChangeDirection, 
    Boolean isUsedButtonSlide)
{
    // 1. 获取技能实例
    Skill skill = GetSkill(skillIndex);
    if (skill == null) return false;
    
    // 2. 检查是否可用
    UsableResult result = skill.isUsable(isDualSkill, withoutStateCheck);
    if (result != UsableResult.kSuccess)
    {
        // 处理不可用原因
        HandleUnusableResult(result);
        return false;
    }
    
    // 3. 尝试使用技能
    if (!skill.TryUseSkill())
        return false;
    
    // 4. 触发技能开始
    skill.onStartSkill(forceChangeDirection, isDualSkill);
    
    // 5. 开始冷却
    startSkillCoolTime(skillIndex, skillLevel, specificCoolTime);
    
    return true;
}
```

### 5.3 isUsable() 检查（line 389587）

```csharp
public virtual UsableResult isUsable(Boolean isDualSkill, Boolean withoutStateCheck)
{
    // 1. 技能等级检查
    if (getLevel() <= 0)
        return UsableResult.kSkillLevelZero;
    
    // 2. 冷却检查
    if (IsInCooltime())
        return UsableResult.kCoolTime;
    
    // 3. MP检查
    if (!isUsableMP(isDualSkill))
        return UsableResult.kLackMp;
    
    // 4. 消耗品检查
    if (CheckUseItem(isDualSkill) < 0)
        return UsableResult.kLackUseItem;
    
    // 5. 状态检查
    if (!withoutStateCheck && !isExcutableState())
        return UsableResult.kExecuteStateFail;
    
    // 6. 使用次数检查
    if (IsLimitUseCountReached())
        return UsableResult.LimitUseCount;
    
    return UsableResult.kSuccess;
}
```

### 15.1 简化技能基类

```csharp
public enum SkillType
{
    Active,     // 主动技能
    Passive,    // 被动技能
    Buff,       // Buff技能
}

public class SimpleSkill
{
    // 基础信息
    public int SkillId;
    public string Name;
    public SkillType Type;
    public int Level;
    public int MaxLevel = 10;
    
    // 消耗
    public int MpCost;
    public float Cooldown;
    public float CurrentCooldown;
    
    // 控制
    public bool IsChargable;
    public bool IsSustainable;
    public int MaxStack;
    public int CurrentStack;
    
    // 状态
    public bool IsUsing;
    public bool IsFinished;
    
    // 关联
    public CharacterBase Owner;
    
    // 回调
    public Action<SimpleSkill> OnStart;
    public Action<SimpleSkill, float> OnUpdate;
    public Action<SimpleSkill> OnEnd;
    public Action<int> OnEnterFrame;
    
    // 生命周期
    public virtual bool CanUse()
    {
        if (Level <= 0) return false;
        if (CurrentCooldown > 0) return false;
        if (Owner.CurrentMP < MpCost) return false;
        return true;
    }
    
    public virtual void Start()
    {
        if (!CanUse()) return;
        
        IsUsing = true;
        IsFinished = false;
        Owner.CurrentMP -= MpCost;
        
        OnStart?.Invoke(this);
    }
    
    public virtual void Update(float deltaTime)
    {
        if (!IsUsing) return;
        
        // 更新冷却
        if (CurrentCooldown > 0)
        {
            CurrentCooldown -= deltaTime;
        }
        
        OnUpdate?.Invoke(this, deltaTime);
    }
    
    public virtual void End()
    {
        IsUsing = false;
        IsFinished = true;
        CurrentCooldown = Cooldown;
        
        OnEnd?.Invoke(this);
    }
    
    public virtual void EnterFrame(int frameIndex)
    {
        OnEnterFrame?.Invoke(frameIndex);
    }
}
```

### 15.2 简化技能管理器

```csharp
public class SimpleSkillManager
{
    private List<SimpleSkill> _skills = new List<SimpleSkill>();
    private CharacterBase _owner;
    
    public SimpleSkillManager(CharacterBase owner)
    {
        _owner = owner;
    }
    
    // 添加技能
    public void AddSkill(SimpleSkill skill)
    {
        skill.Owner = _owner;
        _skills.Add(skill);
    }
    
    // 使用技能
    public bool UseSkill(int skillId)
    {
        var skill = _skills.Find(s => s.SkillId == skillId);
        if (skill == null || !skill.CanUse())
            return false;
        
        skill.Start();
        return true;
    }
    
    // 每帧更新
    public void Update(float deltaTime)
    {
        foreach (var skill in _skills)
        {
            skill.Update(deltaTime);
            
            if (skill.IsUsing && skill.IsFinished)
            {
                skill.End();
            }
        }
    }
    
    // 查询
    public SimpleSkill GetSkill(int skillId) => _skills.Find(s => s.SkillId == skillId);
    public List<SimpleSkill> GetAllSkills() => _skills;
    public List<SimpleSkill> GetActiveSkills() => _skills.FindAll(s => s.Type == SkillType.Active);
    public List<SimpleSkill> GetBuffSkills() => _skills.FindAll(s => s.Type == SkillType.Buff);
}
```

### 15.3 简化射击技能

```csharp
public class SimpleShootSkill : SimpleSkill
{
    public GameObject BulletPrefab;
    public int BulletCount = 1;
    public float BulletSpeed = 10f;
    public int Damage = 100;
    public float ShootInterval = 0.1f;
    
    private float _shootTimer;
    private int _shotCount;
    
    public override void Start()
    {
        base.Start();
        _shootTimer = 0f;
        _shotCount = 0;
    }
    
    public override void Update(float deltaTime)
    {
        base.Update(deltaTime);
        
        if (!IsUsing) return;
        
        _shootTimer += deltaTime;
        
        if (_shootTimer >= ShootInterval && _shotCount < BulletCount)
        {
            Shoot();
            _shootTimer = 0f;
            _shotCount++;
        }
        
        if (_shotCount >= BulletCount)
        {
            IsFinished = true;
        }
    }
    
    private void Shoot()
    {
        var bullet = GameObject.Instantiate(BulletPrefab);
        bullet.transform.position = Owner.transform.position;
        
        var bulletComp = bullet.GetComponent<SimpleBullet>();
        bulletComp.Damage = Damage * Level;  // 伤害随等级提升
        bulletComp.Speed = BulletSpeed;
        bulletComp.Direction = Owner.IsFacingRight ? Vector2.right : Vector2.left;
    }
}
```

## 十六、设计要点总结

### 16.1 核心设计模式

| 模式 | 应用 |
|------|------|
| 模板方法模式 | Skill 基类定义生命周期骨架，子类实现具体行为 |
| 状态模式 | 技能状态集合管理，状态切换触发不同逻辑 |
| 策略模式 | SkillControlType 不同类型有不同的处理策略 |
| 观察者模式 | SkillCallback 回调结构，事件通知 |
| 对象池模式 | 技能实例复用，减少 GC |

### 16.2 关键设计决策

| 决策点 | 设计选择 | 原因 |
|--------|----------|------|
| 技能实例化 | 每个角色独立实例 | 不同角色技能等级、状态独立 |
| 冷却管理 | 技能内部管理 | 简化逻辑，支持联动冷却 |
| 控制类型 | 枚举 + 子类型 | 支持多种输入方式 |
| 动画事件 | 帧号回调 | 精确控制攻击框、特效时机 |
| 滑动连接 | 方向标志 + 技能列表 | 灵活配置技能连招 |

### 16.3 性能优化

- **缓存数据**：Cached_CurrentLevel、Cached_SpendMP 等避免频繁计算
- **预加载资源**：RegistPreloadResoruce 提前加载特效、音效
- **状态集合**：HashSet 快速判断状态有效性
- **冷却时间复用**：联动冷却共享冷却计时器
